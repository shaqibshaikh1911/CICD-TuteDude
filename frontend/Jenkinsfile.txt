pipeline {
    agent any

    tools {
        // This uses the NodeJS tool we configured named 'node-18'
        nodejs 'node-18'
    }

    stages {
        stage('Clone Repository') {
            steps {
                checkout scm
            }
        }

        stage('Install Dependencies') {
            steps {
                sh 'npm install'
            }
        }

        stage('Restart Application with PM2') {
            steps {
                sh '''
                    # Restart the app named 'express-frontend'
                    pm2 restart ecosystem.config.js --only frontend --update-env || pm2 start ecosystem.config.js --only express-frontend
                '''
            }
        }
    }

    post {
        always {
            cleanWs()
        }
        success {
            echo 'Express frontend deployed successfully!'
        }
        failure {
            echo 'Express frontend deployment failed!'
        }
    }
}
